\begin{frame}
    \frametitle{GP Form}

        \begin{align*}
            \text{minimize } &g_0(x) \\
            \text{subject to } &f_i(x) = 1, &i &= 1,\dots,m\\
                               &g_i(x) \leq 1, &i &= 1,\dots,n
        \end{align*}

        \begin{align*}
            f(\bold{x}) &= c x_1^{a_1} x_2^{a_2} \dotsm x_n^{a_n} , \\
            g(\bold{x}) &= \displaystyle\sum_{k=1}^K c_k x_1^{a_{1_k}} x_2^{a_{2_k}} \dotsm x_n^{a_{n_k}}.
        \end{align*}

\end{frame}
